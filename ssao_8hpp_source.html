331  void generateKernel (void)
332  {
333  Eigen::Vector2f sample;
334  kernel = new float[numberOfSamples * 2];
335 
336  float step = 2.0*M_PI/(float)numberOfSamples;
337 
338  // divide in numberOfSamples directions in the unit circle, and multiply each vector by a random radius
339  for (int i = 1; i <= numberOfSamples; i++)
340  {
341  float r = pow(random(0.01, 1.0f),1.0);
342 
343  sample[0] = cos(step*i);
344  sample[1] = sin(step*i);
345 
346  kernel[i*2+0] = r*sample[0];
347  kernel[i*2+1] = r*sample[1];
348  }
349  }
