338  void generateKernel (void)
339  {
340  Eigen::Vector2f sample;
341  kernel = new float[numberOfSamples * 2];
342 
343  float step = 2.0*M_PI/(float)numberOfSamples;
344 
345  // divide in numberOfSamples directions in the unit circle, and multiply each vector by a random radius
346  for (int i = 1; i <= numberOfSamples; i++)
347  {
348  float r = pow(random(0.01, 1.0f),1.0);
349 
350  sample[0] = cos(step*i);
351  sample[1] = sin(step*i);
352 
353  kernel[i*2+0] = r*sample[0];
354  kernel[i*2+1] = r*sample[1];
355  }
356  }
