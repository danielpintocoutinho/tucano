283  void generateKernel (void)
284  {
285  float scale;
286  Eigen::Vector3f sample;
287  kernel = new float[numberOfSamples*3];
288 
289  int temp = 0;
290  for (int i = 0; i<numberOfSamples; i++)
291  {
292  sample = Eigen::Vector3f( random(-1.0f,1.0f) , random(-1.0f,1.0f) , random(0.0f,1.0f) );
293  sample.normalize();
294  //cout << "Kernel: " << sample.transpose() << endl;
295  sample *= random(0.0f,1.0f); //Distribute sample points randomly around the kernel.
296  scale = float(i)/float(numberOfSamples);
297  scale = lerp(0.1f, 1.0f, scale * scale);//Cluster sample points towards origin.
298  sample *= scale;
299  kernel[temp] = sample[0];
300  kernel[temp+1] = sample[1];
301  kernel[temp+2] = sample[2];
302  temp += 3;
303  }
304  }
