333 
334  numberOfVertices = vert.size();

336  // creates new attribute and load vertex coordinates
337  createAttribute("in_Position", vert);

339  // from now on we are just computing some information about the model such as bounding box, centroid ...

342

345

352  }
353 
354  //Y:

357  }

359  yMin = vert[temp][1];
360  }
361 
362  //Z:

364  zMax = vert[temp][2];
365  }
366  if(vert[temp][2] < zMin) {
367  zMin = vert[temp][2];
368  }
369 
370  //W:
371  temp++;
372  }
373 
374  scale = 1.0/max(max(xMax-xMin, yMax-yMin), zMax-zMin);
375

377  float centerY = (yMax+yMin)/2.0;
378  float centerZ = (zMax+zMin)/2.0;
379  objectCenter = Eigen::Vector3f(centerX, centerY, centerZ);
380

383  for(unsigned int i = 0; i < numberOfVertices; i++) {
