409  static Eigen::Matrix4f createPerspectiveMatrix (float fy, float in_aspect_ratio, float in_near_plane,float in_far_plane)
410  {
411  Eigen::Matrix4f out = Eigen::Matrix4f::Zero();
412 
413  const float
414  y_scale = (float)1.0/tan((fy/2.0)*(M_PI/180.0)),
415  x_scale = y_scale / in_aspect_ratio,
416  frustum_length = in_far_plane - in_near_plane;
417 
418  out(0,0) = x_scale;
419  out(1,1) = y_scale;
420  out(2,2) = -((in_far_plane + in_near_plane) / frustum_length);
421  out(3,2) = -1.0;
422  out(2,3) = -((2 * in_near_plane * in_far_plane) / frustum_length);
423 
424  return out;
425  }
