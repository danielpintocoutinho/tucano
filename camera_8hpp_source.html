367  static Eigen::Matrix4f createPerspectiveMatrix (float fy, float in_aspect_ratio, float in_near_plane,float in_far_plane)
368  {
369  Eigen::Matrix4f out = Eigen::Matrix4f::Zero();
370 
371  const float
372  y_scale = (float)1.0/tan((fy/2.0)*(M_PI/180.0)),
373  x_scale = y_scale / in_aspect_ratio,
374  frustum_length = in_far_plane - in_near_plane;
375 
376  out(0,0) = x_scale;
