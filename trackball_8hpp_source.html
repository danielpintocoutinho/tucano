264  Eigen::Vector3f computeSpherePosition (const Eigen::Vector2f& pos)
265  {
266  //In order to handle points outside the sphere, we will use two surfaces: A Sphere with radius = 0.8 and a Hyperbolic Sheet. More information about this usage in: http://www.opengl.org/wiki/Trackball.
267 
268  float z;
269 
270  //If pos corresponds to a point before the intersection of the two surfaces, z can be calculated by the Sphere's equation.
271  if(pos[0]*pos[0] + pos[1]*pos[1] <= (radius*radius)/2.0) {
272  z = sqrt(radius*radius - pos[0]*pos[0] - pos[1]*pos[1]);
273  }
274 
275  //Else, it will be calculated by the Hyperbolic Sheet's equation.
276  else {
277  z = (radius*radius)/(2.0*sqrt(pos[0]*pos[0] + pos[1]*pos[1]));
278  }
279 
280  Eigen::Vector3f position = Eigen::Vector3f(pos[0], pos[1], z);
281 
282  return position;
283  }

377  void computeRotationAngle (void)
378  {
379  //Given two position vectors, corresponding to the initial and final mouse coordinates, calculate the rotation of the sphere that will keep the mouse always in the initial position.
380 
381  if(initialPosition.norm() > 0) {
382  initialPosition.normalize();
383  }
384  if(finalPosition.norm() > 0) {
385  finalPosition.normalize();
386  }
387 
388  //cout << "Initial Position: " << initialPosition.transpose() << " Final Position: " << finalPosition.transpose() << endl << endl;
389 
390  Eigen::Vector3f rotationAxis = initialPosition.cross(finalPosition);
391 
392  if(rotationAxis.norm() != 0) {
393  rotationAxis.normalize();
394  }
395 
396  float dot = initialPosition.dot(finalPosition);
397 
398  float rotationAngle = (dot <= 1) ? acos(dot) : 0;//If, by losing floating point precision, the dot product between the initial and final positions is bigger than one, ignore the rotation.
399 
400  Eigen::Quaternion<float> q (Eigen::AngleAxis<float>(rotationAngle,rotationAxis));
401 
402  quaternion = q * quaternion;
403  quaternion.normalize();
404  }
