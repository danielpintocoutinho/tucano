259  Eigen::Vector3f computeSpherePosition (const Eigen::Vector2f& pos)
260  {
261  //In order to handle points outside the sphere, we will use two surfaces: A Sphere with radius = 0.8 and a Hyperbolic Sheet. More information about this usage in: http://www.opengl.org/wiki/Trackball.
262 
263  float z;
264 
265  //If pos corresponds to a point before the intersection of the two surfaces, z can be calculated by the Sphere's equation.
266  if(pos[0]*pos[0] + pos[1]*pos[1] <= (radius*radius)/2.0) {
267  z = sqrt(radius*radius - pos[0]*pos[0] - pos[1]*pos[1]);
268  }
269 
270  //Else, it will be calculated by the Hyperbolic Sheet's equation.
271  else {
272  z = (radius*radius)/(2.0*sqrt(pos[0]*pos[0] + pos[1]*pos[1]));
273  }
274 
275  Eigen::Vector3f position = Eigen::Vector3f(pos[0], pos[1], z);
276 
277  return position;
278  }

370  void computeRotationAngle (void)
371  {
372  //Given two position vectors, corresponding to the initial and final mouse coordinates, calculate the rotation of the sphere that will keep the mouse always in the initial position.
373 
374  if(initialPosition.norm() > 0) {
375  initialPosition.normalize();
376  }
377  if(finalPosition.norm() > 0) {
378  finalPosition.normalize();
379  }
380 
381  //cout << "Initial Position: " << initialPosition.transpose() << " Final Position: " << finalPosition.transpose() << endl << endl;
382 
383  Eigen::Vector3f rotationAxis = initialPosition.cross(finalPosition);
384 
385  if(rotationAxis.norm() != 0) {
386  rotationAxis.normalize();
387  }
388 
389  float dot = initialPosition.dot(finalPosition);
390 
391  float rotationAngle = (dot <= 1) ? acos(dot) : 0;//If, by losing floating point precision, the dot product between the initial and final positions is bigger than one, ignore the rotation.
392 
393  Eigen::Quaternion<float> q (Eigen::AngleAxis<float>(rotationAngle,rotationAxis));
394 
395  quaternion = q * quaternion;
396  quaternion.normalize();
397  }
