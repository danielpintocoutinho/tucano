641  ply->buffer_last = size;
642  ply->buffer_first = ply->buffer_token = 0;
643  /* fill remaining with new data */
644  size = fread(ply->buffer+size, 1, BUFFERSIZE-size-1, ply->fp);
645  /* place sentinel so we can use str* functions with buffer */
646  ply->buffer[BUFFERSIZE-1] = '\0';
647  /* check if read failed */
648  if (size <= 0) return 0;
649  /* increase size to account for new data */
650  ply->buffer_last += size;
651  return 1;
652 }

654 /* We don't care about end-of-line, generally, because we
655  * separate words by any white-space character.
656  * Unfortunately, in binary mode, right after 'end_header',
657  * we have to know *exactly* how many characters to skip */
658 /* We use the end-of-line marker after the 'ply' magic
659  * number to figure out what to do */
660 static int ply_read_header_magic(p_ply ply) {
661  char *magic = ply->buffer;
662  if (!BREFILL(ply)) {
663  ply->error_cb(ply, "Unable to read magic number from file");
664  return 0;
665  }
666  /* check if it is ply */
667  if (magic[0] != 'p' || magic[1] != 'l' || magic[2] != 'y'
668  || !isspace(magic[3])) {
669  ply->error_cb(ply, "Wrong magic number. Expected 'ply'");
670  return 0;
671  }
672  /* figure out if we have to skip the extra character
673  * after header when we reach the binary part of file */
674  ply->rn = magic[3] == '\r' && magic[4] == '\n';
675  BSKIP(ply, 3);
676  return 1;
677 }

987  assert(ply && ply->fp && ply->io_mode == PLY_READ);
988  if (!ply_read_header_magic(ply)) return 0;
989  if (!ply_read_word(ply)) return 0;

1006  if (BSIZE(ply) < 1 && !BREFILL(ply)) {
1007  ply_ferror(ply, "Unexpected end of file");

1011  }

1535 static int ply_read_word(p_ply ply) {
1536  size_t t = 0;
1537  assert(ply && ply->fp && ply->io_mode == PLY_READ);
1538  /* skip leading blanks */
1539  while (1) {

1547  } else break;

1553  if (t < BSIZE(ply)) {

1556  }
1557  /* otherwise, try to refill buffer */
1558  if (!BREFILL(ply)) {
1559  /* if we reached the end of file, try to do with what we have */
1560  ply_finish_word(ply, t);
1561  return ply_check_word(ply);

1563  /* return 0; */
1564  }
1565  /* keep looking from where we left */
1566  t += strcspn(BFIRST(ply) + t, " \n\r\t");
1567  /* check if the token is too large for our buffer */
1568  if (t >= BSIZE(ply)) {
1569  ply_ferror(ply, "Token too large");

1571  }
